START

SET database reference to "attendance"

WHILE TRUE
    DISPLAY Main Menu:
        1. Display Students
        2. Add Student
        3. Update Student
        4. Delete Student
        5. Features
        6. Exit

    INPUT choice

    IF choice = 1 THEN
        GET all students from database
        IF data exists THEN
            FOR EACH student
                DISPLAY ID, Name, and Dates with Status
            END FOR
        ELSE
            DISPLAY "No students found"
        END IF

    ELSE IF choice = 2 THEN
        INPUT student ID and name
        CREATE empty date list

        REPEAT
            INPUT date (or "done")
            IF "done" THEN EXIT loop
            INPUT status
            ADD (date, status) to list
        UNTIL done

        SAVE student to database
        DISPLAY "Student added"

    ELSE IF choice = 3 THEN
        INPUT student ID
        GET student data

        IF student exists THEN
            DISPLAY current details

            INPUT new name (optional)
            IF given THEN UPDATE name

            ASK if update dates
            IF yes THEN
                CREATE new date list
                REPEAT date input process
                UPDATE dates
            END IF

            DISPLAY "Student updated"
        ELSE
            DISPLAY "Student not found"
        END IF

    ELSE IF choice = 4 THEN
        INPUT student ID
        GET student data

        IF student exists THEN
            ASK confirmation
            IF yes THEN
                DELETE student
                DISPLAY "Deleted"
            ELSE
                DISPLAY "Cancelled"
            END IF
        ELSE
            DISPLAY "Student not found"
        END IF

    ELSE IF choice = 5 THEN
        WHILE TRUE
            DISPLAY Features Menu:
                1. Lowest Attendance Dates
                2. Highest Attendance Dates
                3. Average Absents per Day
                6. Back

            INPUT feature_choice

            IF feature_choice = 1 THEN
                COUNT "Present" per date
                FIND minimum
                DISPLAY lowest attendance dates

            ELSE IF feature_choice = 2 THEN
                COUNT "Present" per date
                FIND maximum
                DISPLAY highest attendance dates

            ELSE IF feature_choice = 3 THEN
                COUNT absences (Absent, Tardy, Cut Class)
                COUNT unique dates
                COMPUTE average = total / dates
                DISPLAY result

            ELSE IF feature_choice = 6 THEN
                BREAK (return to main menu)

            ELSE
                DISPLAY "Invalid choice"
            END IF
        END WHILE

    ELSE IF choice = 6 THEN
        EXIT program

    ELSE
        DISPLAY "Invalid choice"
    END IF

END WHILE

END
